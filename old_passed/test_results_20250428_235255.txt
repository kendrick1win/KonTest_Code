from typing import List

def has_close_elements(numbers: List[float], threshold: float) -> bool:
    if len(numbers) < 2:
        return False
    i = 0
    while i < len(numbers) - 1:
        if numbers[i + 1] - numbers[i] <= threshold:
            return True
        i += 1
    return False
